    <script>
        const grammarData = [
            {
                title: "Simple Sentences",
                theory: "A simple sentence consists of one independent clause. It has a subject and a predicate.",
                example: "Example: 'She writes every day.'",
                question: "Which of these is a simple sentence?",
                options: [
                    { text: "The dog barked.", correct: true },
                    { text: "The dog barked and the cat ran.", correct: false },
                    { text: "Because the dog barked.", correct: false }
                ],
                explanation: "Correct! 'The dog barked' has one subject (dog) and one verb (barked)."
            },
            {
                title: "Compound Sentences",
                theory: "A compound sentence joins two independent clauses together, usually with a conjunction like 'and', 'but', or 'or'.",
                example: "Example: 'I cooked dinner, and he washed the dishes.'",
                question: "Which sentence uses a coordinating conjunction (FANBOYS)?",
                options: [
                    { text: "I like coffee.", correct: false },
                    { text: "I like coffee, but Mary likes tea.", correct: true },
                    { text: "Although I like coffee.", correct: false }
                ],
                explanation: "Great! 'But' is a conjunction used to connect two independent ideas."
            },
            {
                title: "Complex Sentences",
                theory: "A complex sentence has one independent clause and at least one dependent (subordinate) clause.",
                example: "Example: 'Because it was raining, we stayed inside.'",
                question: "Identify the complex sentence:",
                options: [
                    { text: "He ran fast.", correct: false },
                    { text: "He ran fast, yet he missed the bus.", correct: false },
                    { text: "After he ran the race, he felt tired.", correct: true }
                ],e ran the race' is a dependent clause."
            }
        ];

        let currentModule = 0;

        function loadModule(index) {
            currentModule = index;
            const data = grammarData[index];
            
            // Update Menu UI
            document.querySelectorAll('.menu-item').forEach((item, idx) => {
                item.classList.toggle('active', idx === index);
            });

            // Update Content
            document.getElementById('title').innerText = data.title;
            document.getElementById('theory-text').innerText = data.theory;
            document.getElementById('example-text').innerText = data.example;
            document.getElementById('question-text').innerText = data.question;

            // Update Quiz
            const optionsContainer = document.getElementById('options-container');
            optionsContainer.innerHTML = '';
            data.options.forEach(opt => {
                const btn = document.createElement('button');
                btn.className = 'quiz-option';
                btn.innerText = opt.text;
                btn.onclick = () => checkAnswer(opt.correct, data.explanation);
                optionsContainer.appendChild(btn);
            });

            // Reset feedback
            const feedback = document.getElementById('feedback');
            feedback.style.display = 'none';
            document.getElementById('next-btn').style.display = 'none';
        }

        function checkAnswer(isCorrect, msg) {
            const feedback = document.getElementById('feedback');
            feedback.style.display = 'block';
            if (isCorrect) {
                feedback.className = 'correct';
                feedback.innerText = msg;
                document.getElementById('next-btn').style.display = 'block';
            } else {
                feedback.className = 'incorrect';
                feedback.innerText = "Try again! Look for the specific pattern mentioned in the theory.";
            }
        }

        function nextQuestion() {
            if (currentModule < grammarData.length - 1) {
                loadModule(currentModule + 1);
            } else {
                alert("Congratulations! You've completed the prototype path.");
            }
        }

        // Initialize first module
        loadModule(0);
    </script>
